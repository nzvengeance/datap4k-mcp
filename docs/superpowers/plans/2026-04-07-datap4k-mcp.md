# datap4k-mcp Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Rust MCP server that indexes extracted Star Citizen p4k data into a hybrid SQLite + Cozo store and exposes search, lookup, traversal, path-finding, and diffing tools to LLMs.

**Architecture:** Plugin parser system normalizes scdatatools JSON and unp4k XML into a Common Entity Model (nodes + edges). SQLite handles search/lookup/bulk via FTS5. Cozo handles graph traversal and path discovery via Datalog. The MCP layer (rmcp 1.3+) exposes 11 tools, 4 resources, and 4 prompts over stdio transport. LLM-guided init flow handles first-run configuration.

**Tech Stack:** Rust, rmcp (official MCP SDK), rusqlite (SQLite + FTS5), cozo (embedded Datalog), serde/serde_json, quick-xml, tokio, clap, toml, walkdir, tracing

**Spec:** `docs/superpowers/specs/2026-04-07-datap4k-mcp-design.md`

```rust
/// A node in the entity graph.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Node {
    pub id: uuid::Uuid,
    pub class_name: String,
    pub record_name: String,
    pub entity_type: EntityType,
    pub source: String,
    pub source_path: String,
    pub game_version: String,
    pub properties: HashMap<String, serde_json::Value>,
}

/// An edge (relationship) between two entities.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Edge {
    pub source_id: uuid::Uuid,
    pub target_id: uuid::Uuid,
    pub label: String,
    pub source_field: String,
    pub properties: HashMap<String, serde_json::Value>,
}

/// A warning emitted during parsing (non-fatal).
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ParseWarning {
    pub source_path: String,
    pub message: String,
}

/// The result of parsing a data directory.
#[derive(Debug, Clone, Default)]
pub struct ParseResult {
    pub nodes: Vec<Node>,
    pub edges: Vec<Edge>,
    pub warnings: Vec<ParseWarning>,
}

impl ParseResult {
    pub fn new() -> Self {
        Self::default()
    }

    /// Merge another ParseResult into this one.
    pub fn merge(&mut self, other: ParseResult) {
        self.nodes.extend(other.nodes);
        self.edges.extend(other.edges);
        self.warnings.extend(other.warnings);
    }
}

/// Extract version code and build number from a p4k directory name.
///
/// Expects format like "4.7.0-live.11518367" → ("4.7.0-live", Some("11518367"))
pub fn version_from_dirname(dirname: &str) -> Option<(String, Option<String>)> {
    // Pattern: major.minor.patch-channel.build
    let re_pattern = regex_lite::Regex::new(
        r"^(\d+\.\d+\.\d+-[a-z]+)(?:\.(\d+))?$"
    ).ok()?;

    let caps = re_pattern.captures(dirname)?;
    let version = caps.get(1)?.as_str().to_string();
    let build = caps.get(2).map(|m| m.as_str().to_string());
    Some((version, build))
}
```

```rust
use anyhow::Result;
use serde::{Deserialize, Serialize};
use std::path::Path;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Config {
    #[serde(default)]
    pub sources: Sources,
    #[serde(default)]
    pub index: IndexConfig,
    #[serde(default)]
    pub server: ServerConfig,
}

#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct Sources {
    #[serde(default)]
    pub versions: Vec<VersionSource>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct VersionSource {
    pub path: String,
    pub version: String,
    #[serde(default = "default_parser")]
    pub parser: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct IndexConfig {
    #[serde(default = "default_index_path")]
    pub path: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ServerConfig {
    #[serde(default = "default_log_level")]
    pub log_level: String,
}

fn default_parser() -> String { "auto".to_string() }
fn default_log_level() -> String { "info".to_string() }

fn default_index_path() -> String {
    dirs::home_dir()
        .map(|h| h.join(".datap4k-mcp").join("index").to_string_lossy().to_string())
        .unwrap_or_else(|| ".datap4k-mcp/index".to_string())
}

impl Default for Config {
    fn default() -> Self {
        Self {
            sources: Sources::default(),
            index: IndexConfig { path: default_index_path() },
            server: ServerConfig { log_level: default_log_level() },
        }
    }
}

impl Default for IndexConfig {
    fn default() -> Self {
        Self { path: default_index_path() }
    }
}

impl Default for ServerConfig {
    fn default() -> Self {
        Self { log_level: default_log_level() }
    }
}

impl Config {
    /// Load config from a file path. Returns default config if file doesn't exist.
    pub fn load_from(path: &Path) -> Result<Self> {
        if !path.exists() {
            return Ok(Self::default());
        }
        let content = std::fs::read_to_string(path)?;
        let config: Config = toml::from_str(&content)?;
        Ok(config)
    }

    /// Load config from the default location (~/.datap4k-mcp/config.toml).
    pub fn load() -> Result<Self> {
        let path = Self::default_path();
        Self::load_from(&path)
    }

    /// Save config to a file path.
    pub fn save_to(&self, path: &Path) -> Result<()> {
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)?;
        }
        let content = toml::to_string_pretty(self)?;
        std::fs::write(path, content)?;
        Ok(())
    }

    /// Save config to the default location.
    pub fn save(&self) -> Result<()> {
        self.save_to(&Self::default_path())
    }

    /// Default config file path.
    pub fn default_path() -> std::path::PathBuf {
        dirs::home_dir()
            .unwrap_or_else(|| std::path::PathBuf::from("."))
            .join(".datap4k-mcp")
            .join("config.toml")
    }

    /// Add or update a version source. If the version already exists, updates its path and parser.
    pub fn add_version(&mut self, path: &str, version: &str, parser: &str) {
        if let Some(existing) = self.sources.versions.iter_mut().find(|v| v.version == version) {
            existing.path = path.to_string();
            existing.parser = parser.to_string();
        } else {
            self.sources.versions.push(VersionSource {
                path: path.to_string(),
                version: version.to_string(),
                parser: parser.to_string(),
            });
        }
    }
}
```

```rust
/// Trait for p4k data format parsers.

    /// Human-readable name of this parser.

    fn detect(&self, path: &Path) -> bool;

    fn parse(&self, path: &Path, version: &str) -> Result<ParseResult>;
}

pub use self::scdatatools::ScdatatoolsParser;
pub use self::unp4k::Unp4kParser;

    vec![

        Box::new(Unp4kParser),
    ]
}

/// Auto-detect which parsers can handle a directory and return them.

    all_parsers().into_iter().filter(|p| p.detect(path)).collect()
```

`src/parser/scdatatools.rs` — this is the largest single file. It handles:
- Walking `DataCore/libs/foundry/records/` for JSON files
- Extracting `_RecordName_`, `_RecordId_`, `_RecordValue_` from each
- Classifying entity type from `SAttachableComponentParams.AttachDef.Type`
- Finding `file://` references recursively in the JSON tree → edges
- Walking `Extracted/Loadouts_Character_Converted/` for NPC loadout XMLs
- Walking `Extracted/SOC_Unpacked/` for object container XMLs

```rust
use crate::model::*;
use anyhow::{Context, Result};
use std::collections::HashMap;
use std::path::Path;
use walkdir::WalkDir;

pub struct ScdatatoolsParser;

impl super::P4kParser for ScdatatoolsParser {
    fn name(&self) -> &str { "scdatatools" }

    fn detect(&self, path: &Path) -> bool {
        // scdatatools output has DataCore/libs/foundry/records/
        path.join("DataCore/libs/foundry/records").is_dir()
    }

    fn parse(&self, path: &Path, version: &str) -> Result<ParseResult> {
        let mut result = ParseResult::new();

        // Phase 1: DataCore JSON records
        let datacore_dir = path.join("DataCore/libs/foundry/records");
        if datacore_dir.is_dir() {
            self.parse_datacore(&datacore_dir, path, version, &mut result)?;
        }

        // Phase 2: Character loadouts (XML)
        let loadouts_dir = path.join("Extracted/Loadouts_Character_Converted");
        if loadouts_dir.is_dir() {
            self.parse_loadouts(&loadouts_dir, path, version, &mut result)?;
        }

        // Phase 3: SOC Object Containers (XML)
        let soc_dir = path.join("Extracted/SOC_Unpacked");
        if soc_dir.is_dir() {
            self.parse_soc(&soc_dir, path, version, &mut result)?;
        }

        tracing::info!(
            "scdatatools parser: {} nodes, {} edges, {} warnings",
            result.nodes.len(),
            result.edges.len(),
            result.warnings.len()
        );
        Ok(result)
    }
}

impl ScdatatoolsParser {
    fn parse_datacore(
        &self,
        datacore_dir: &Path,
        base_path: &Path,
        version: &str,
        result: &mut ParseResult,
    ) -> Result<()> {
        for entry in WalkDir::new(datacore_dir)
            .into_iter()
            .filter_map(|e| e.ok())
            .filter(|e| e.path().extension().map_or(false, |ext| ext == "json"))
        {
            match self.parse_datacore_file(entry.path(), base_path, version) {
                Ok((node, edges)) => {
                    result.nodes.push(node);
                    result.edges.extend(edges);
                }
                Err(e) => {
                    result.warnings.push(ParseWarning {
                        source_path: entry.path().strip_prefix(base_path)
                            .unwrap_or(entry.path())
                            .to_string_lossy()
                            .to_string(),
                        message: format!("{e:#}"),
                    });
                }
            }
        }
        Ok(())
    }

    fn parse_datacore_file(
        &self,
        path: &Path,
        base_path: &Path,
        version: &str,
    ) -> Result<(Node, Vec<Edge>)> {
        let content = std::fs::read_to_string(path)
            .with_context(|| format!("reading {}", path.display()))?;
        let json: serde_json::Value = serde_json::from_str(&content)
            .with_context(|| format!("parsing JSON {}", path.display()))?;

        let record_name = json.get("_RecordName_")
            .and_then(|v| v.as_str())
            .unwrap_or("")
            .to_string();

        let record_id = json.get("_RecordId_")
            .and_then(|v| v.as_str())
            .unwrap_or("");

        let id = uuid::Uuid::parse_str(record_id)
            .unwrap_or_else(|_| uuid::Uuid::new_v4());

        // Extract class_name from record_name (after the last dot, or the full thing)
        let class_name = record_name
            .rsplit_once('.')
            .map(|(_, name)| name)
            .unwrap_or(&record_name)
            .to_string();

        let record_value = json.get("_RecordValue_")
            .cloned()
            .unwrap_or(serde_json::Value::Null);

        // Classify entity type
        let entity_type = classify_entity_type(&record_value, path);

        // Extract properties (flatten _RecordValue_ into properties)
        let properties = if let serde_json::Value::Object(map) = &record_value {
            map.iter()
                .filter(|(k, _)| !k.starts_with('_'))
                .map(|(k, v)| (k.clone(), v.clone()))
                .collect()
        } else {
            HashMap::new()
        };

        let source_path = path.strip_prefix(base_path)
            .unwrap_or(path)
            .to_string_lossy()
            .to_string();

        let node = Node {
            id,
            class_name,
            record_name,
            entity_type,
            source: "scdatatools".to_string(),
            source_path,
            game_version: version.to_string(),
            properties,
        };

        // Extract edges from file:// references
        let mut edges = Vec::new();
        extract_file_refs(&record_value, id, "", &mut edges);

        Ok((node, edges))
    }

    fn parse_loadouts(
        &self,
        loadouts_dir: &Path,
        base_path: &Path,
        version: &str,
        result: &mut ParseResult,
    ) -> Result<()> {
        for entry in WalkDir::new(loadouts_dir)
            .into_iter()
            .filter_map(|e| e.ok())
            .filter(|e| e.path().extension().map_or(false, |ext| ext == "xml"))
        {
            if let Err(e) = self.parse_loadout_file(entry.path(), base_path, version, result) {
                result.warnings.push(ParseWarning {
                    source_path: entry.path().strip_prefix(base_path)
                        .unwrap_or(entry.path())
                        .to_string_lossy()
                        .to_string(),
                    message: format!("{e:#}"),
                });
            }
        }
        Ok(())
    }

    fn parse_loadout_file(
        &self,
        path: &Path,
        base_path: &Path,
        version: &str,
        result: &mut ParseResult,
    ) -> Result<()> {
        let content = std::fs::read_to_string(path)?;
        // Basic XML parsing for loadout items
        // Extract itemName references as edges to the loadout node
        let loadout_name = path.file_stem()
            .map(|s| s.to_string_lossy().to_string())
            .unwrap_or_default();

        let loadout_id = uuid::Uuid::new_v4();
        let source_path = path.strip_prefix(base_path)
            .unwrap_or(path)
            .to_string_lossy()
            .to_string();

        result.nodes.push(Node {
            id: loadout_id,
            class_name: loadout_name.clone(),
            record_name: format!("Loadout.{loadout_name}"),
            entity_type: EntityType::Loadout,
            source: "scdatatools".to_string(),
            source_path,
            game_version: version.to_string(),
            properties: HashMap::new(),
        });

        // Extract itemName attributes from XML
        for line in content.lines() {
            if let Some(item_name) = extract_xml_attr(line, "itemName") {
                if !item_name.is_empty() && item_name != "empty" {
                    let port_name = extract_xml_attr(line, "portName").unwrap_or_default();
                    result.edges.push(Edge {
                        source_id: loadout_id,
                        target_id: uuid::Uuid::nil(), // resolved later by class_name lookup
                        label: "equips".to_string(),
                        source_field: format!("itemName:{item_name}"),
                        properties: {
                            let mut p = HashMap::new();
                            p.insert("port_name".to_string(), serde_json::Value::String(port_name));
                            p.insert("item_class_name".to_string(), serde_json::Value::String(item_name));
                            p
                        },
                    });
                }
            }
        }

        Ok(())
    }

    fn parse_soc(
        &self,
        soc_dir: &Path,
        base_path: &Path,
        version: &str,
        result: &mut ParseResult,
    ) -> Result<()> {
        for entry in WalkDir::new(soc_dir)
            .into_iter()
            .filter_map(|e| e.ok())
            .filter(|e| {
                let p = e.path();
                p.extension().map_or(false, |ext| ext == "xml")
                    && !p.to_string_lossy().contains("_editor.xml")
            })
        {
            if let Err(e) = self.parse_soc_file(entry.path(), base_path, version, result) {
                result.warnings.push(ParseWarning {
                    source_path: entry.path().strip_prefix(base_path)
                        .unwrap_or(entry.path())
                        .to_string_lossy()
                        .to_string(),
                    message: format!("{e:#}"),
                });
            }
        }
        Ok(())
    }

    fn parse_soc_file(
        &self,
        path: &Path,
        base_path: &Path,
        version: &str,
        result: &mut ParseResult,
    ) -> Result<()> {
        let content = std::fs::read_to_string(path)?;
        let container_name = path.file_stem()
            .map(|s| s.to_string_lossy().to_string())
            .unwrap_or_default();

        let location_id = uuid::Uuid::new_v4();
        let source_path = path.strip_prefix(base_path)
            .unwrap_or(path)
            .to_string_lossy()
            .to_string();

        result.nodes.push(Node {
            id: location_id,
            class_name: container_name.clone(),
            record_name: format!("ObjectContainer.{container_name}"),
            entity_type: EntityType::Location,
            source: "scdatatools".to_string(),
            source_path,
            game_version: version.to_string(),
            properties: HashMap::new(),
        });

        // Extract entity references from SOC XML
        for line in content.lines() {
            if let Some(entity_class) = extract_xml_attr(line, "entityClass") {
                if !entity_class.is_empty() {
                    result.edges.push(Edge {
                        source_id: location_id,
                        target_id: uuid::Uuid::nil(),
                        label: "contains_entity".to_string(),
                        source_field: format!("entityClass:{entity_class}"),
                        properties: {
                            let mut p = HashMap::new();
                            p.insert("entity_class_name".to_string(),
                                serde_json::Value::String(entity_class));
                            if let Some(pos) = extract_xml_attr(line, "pos") {
                                p.insert("position".to_string(),
                                    serde_json::Value::String(pos));
                            }
                            p
                        },
                    });
                }
            }
        }

        Ok(())
    }
}

/// Classify an entity type from its DataCore record value.
fn classify_entity_type(record_value: &serde_json::Value, path: &Path) -> EntityType {
    // Try SAttachableComponentParams.AttachDef.Type
    if let Some(attach_type) = record_value
        .pointer("/Components/SAttachableComponentParams/AttachDef/Type")
        .and_then(|v| v.as_str())
    {
        return match attach_type {
            "Ship" => EntityType::Ship,
            "Vehicle" | "GroundVehicle" => EntityType::Vehicle,
            "WeaponGun" | "Turret" | "TurretBase" | "MissileLauncher" => EntityType::WeaponShip,
            "WeaponPersonal" | "Knife" => EntityType::WeaponPersonal,
            "Cooler" | "PowerPlant" | "Shield" | "QuantumDrive" | "Radar"
            | "MiningModifier" | "SalvageHead" | "SalvageModifier" => EntityType::Component,
            "Armor" | "Helmet" | "Backpack" | "Undersuit" => EntityType::Armor,
            _ => EntityType::Unknown,
        };
    }

    // Try to classify by _Type_ field
    if let Some(type_field) = record_value.get("_Type_").and_then(|v| v.as_str()) {
        return match type_field {
            "AmmoParams" => EntityType::Ammo,
            "MissionBroker" | "MissionData" => EntityType::Mission,
            "LootGeneration" => EntityType::LootTable,
            t if t.contains("Faction") => EntityType::Faction,
            t if t.contains("Reputation") => EntityType::Reputation,
            t if t.contains("Craft") => EntityType::CraftingBlueprint,
            t if t.contains("Shop") => EntityType::Shop,
            t if t.contains("Audio") || t.contains("Voice") => EntityType::AudioDef,
            _ => EntityType::Unknown,
        };
    }

    // Fallback: classify by file path
    let path_str = path.to_string_lossy().to_lowercase();
    if path_str.contains("/spaceships/") { EntityType::Ship }
    else if path_str.contains("/groundvehicles/") { EntityType::Vehicle }
    else if path_str.contains("/weapons/") { EntityType::WeaponShip }
    else if path_str.contains("/ammoparams/") { EntityType::Ammo }
    else if path_str.contains("/factions") { EntityType::Faction }
    else if path_str.contains("/reputation") { EntityType::Reputation }
    else if path_str.contains("/crafting") { EntityType::CraftingBlueprint }
    else if path_str.contains("/lootgeneration") { EntityType::LootTable }
    else if path_str.contains("/missiondata") || path_str.contains("/missionbroker") { EntityType::Mission }
    else { EntityType::Unknown }
}

/// Recursively extract file:// references from a JSON value, creating edges.
fn extract_file_refs(
    value: &serde_json::Value,
    source_id: uuid::Uuid,
    field_path: &str,
    edges: &mut Vec<Edge>,
) {
    match value {
        serde_json::Value::String(s) if s.starts_with("file://") => {
            // Extract the target record ID from the file path if possible
            // file:// refs point to other DataCore records
            let label = field_path
                .rsplit('/')
                .next()
                .unwrap_or("references")
                .to_string();
            let label = format!("has_{}", label.to_lowercase()
                .replace("record", "")
                .replace("params", "")
                .trim_matches('_'));

            edges.push(Edge {
                source_id,
                target_id: uuid::Uuid::nil(), // resolved later by path lookup
                label: if label == "has_" { "references".to_string() } else { label },
                source_field: field_path.to_string(),
                properties: {
                    let mut p = HashMap::new();
                    p.insert("file_ref".to_string(), serde_json::Value::String(s.clone()));
                    p
                },
            });
        }
        serde_json::Value::Object(map) => {
            // Check for _RecordPath_ pattern (embedded reference object)
            if map.contains_key("_RecordPath_") {
                if let Some(record_id) = map.get("_RecordId_").and_then(|v| v.as_str()) {
                    let target_id = uuid::Uuid::parse_str(record_id)
                        .unwrap_or(uuid::Uuid::nil());
                    let label = field_path
                        .rsplit('/')
                        .next()
                        .unwrap_or("references")
                        .to_string();

                    edges.push(Edge {
                        source_id,
                        target_id,
                        label: format!("ref_{}", label.to_lowercase()),
                        source_field: field_path.to_string(),
                        properties: HashMap::new(),
                    });
                }
                return; // Don't recurse into reference objects
            }

            for (key, val) in map {
                if key.starts_with('_') { continue; } // skip meta fields
                let new_path = if field_path.is_empty() {
                    key.clone()
                } else {
                    format!("{field_path}/{key}")
                };
                extract_file_refs(val, source_id, &new_path, edges);
            }
        }
        serde_json::Value::Array(arr) => {
            for (i, val) in arr.iter().enumerate() {
                let new_path = format!("{field_path}[{i}]");
                extract_file_refs(val, source_id, &new_path, edges);
            }
        }
        _ => {}
    }
}

/// Extract an XML attribute value from a line (simple regex-free extraction).
fn extract_xml_attr(line: &str, attr_name: &str) -> Option<String> {
    let pattern = format!("{attr_name}=\"");
    let start = line.find(&pattern)? + pattern.len();
    let end = line[start..].find('"')? + start;
    Some(line[start..end].to_string())
}
```

```rust
use crate::model::*;
use anyhow::{Context, Result};
use rusqlite::{params, Connection};
use std::path::Path;

pub struct SqliteIndex {
    conn: Connection,
}

#[derive(Debug, Clone)]
pub struct VersionInfo {
    pub id: i64,
    pub code: String,
    pub build_number: Option<String>,
    pub data_path: String,
}

impl SqliteIndex {
    pub fn open(path: &Path) -> Result<Self> {
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)?;
        }
        let conn = Connection::open(path)
            .with_context(|| format!("opening SQLite at {}", path.display()))?;

        let index = Self { conn };
        index.create_schema()?;
        Ok(index)
    }

    pub fn open_in_memory() -> Result<Self> {
        let conn = Connection::open_in_memory()?;
        let index = Self { conn };
        index.create_schema()?;
        Ok(index)
    }

    fn create_schema(&self) -> Result<()> {
        self.conn.execute_batch("
            CREATE TABLE IF NOT EXISTS versions (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                code TEXT NOT NULL UNIQUE,
                build_number TEXT,
                data_path TEXT NOT NULL,
                indexed_at TEXT DEFAULT (datetime('now'))
            );

            CREATE TABLE IF NOT EXISTS sources (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                version_id INTEGER REFERENCES versions(id),
                parser_name TEXT NOT NULL,
                file_count INTEGER DEFAULT 0,
                indexed_at TEXT DEFAULT (datetime('now'))
            );

            CREATE TABLE IF NOT EXISTS entities (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                uuid TEXT NOT NULL,
                class_name TEXT NOT NULL,
                record_name TEXT NOT NULL,
                entity_type TEXT NOT NULL,
                source TEXT NOT NULL,
                source_path TEXT NOT NULL,
                game_version TEXT NOT NULL,
                properties_json TEXT NOT NULL DEFAULT '{}'
            );

            CREATE INDEX IF NOT EXISTS idx_entities_uuid ON entities(uuid);
            CREATE INDEX IF NOT EXISTS idx_entities_class_name ON entities(class_name);
            CREATE INDEX IF NOT EXISTS idx_entities_entity_type ON entities(entity_type);
            CREATE INDEX IF NOT EXISTS idx_entities_game_version ON entities(game_version);

            CREATE VIRTUAL TABLE IF NOT EXISTS entity_fts USING fts5(
                class_name,
                record_name,
                properties_text,
                content='entities',
                content_rowid='id',
                tokenize='porter unicode61'
            );

            CREATE TRIGGER IF NOT EXISTS entities_ai AFTER INSERT ON entities BEGIN
                INSERT INTO entity_fts(rowid, class_name, record_name, properties_text)
                VALUES (new.id, new.class_name, new.record_name, new.properties_json);
            END;
        ")?;
        Ok(())
    }

    pub fn insert_nodes(&self, nodes: &[Node]) -> Result<()> {
        let tx = self.conn.unchecked_transaction()?;
        {
            let mut stmt = tx.prepare_cached(
                "INSERT INTO entities (uuid, class_name, record_name, entity_type, source, source_path, game_version, properties_json)
                 VALUES (?1, ?2, ?3, ?4, ?5, ?6, ?7, ?8)"
            )?;

            for node in nodes {
                let props_json = serde_json::to_string(&node.properties)?;
                stmt.execute(params![
                    node.id.to_string(),
                    node.class_name,
                    node.record_name,
                    node.entity_type.as_str(),
                    node.source,
                    node.source_path,
                    node.game_version,
                    props_json,
                ])?;
            }
        }
        tx.commit()?;
        Ok(())
    }

    pub fn entity_count(&self) -> Result<i64> {
        let count: i64 = self.conn.query_row(
            "SELECT COUNT(*) FROM entities", [], |row| row.get(0)
        )?;
        Ok(count)
    }

    pub fn search(&self, query: &str, limit: usize) -> Result<Vec<Node>> {
        let mut stmt = self.conn.prepare(
            "SELECT e.uuid, e.class_name, e.record_name, e.entity_type, e.source,
                    e.source_path, e.game_version, e.properties_json
             FROM entity_fts f
             JOIN entities e ON e.id = f.rowid
             WHERE entity_fts MATCH ?1
             LIMIT ?2"
        )?;

        let nodes = stmt.query_map(params![query, limit as i64], |row| {
            Ok(row_to_node(row))
        })?.filter_map(|r| r.ok()).collect();

        Ok(nodes)
    }

    pub fn lookup_by_uuid(&self, uuid: &uuid::Uuid) -> Result<Option<Node>> {
        let mut stmt = self.conn.prepare(
            "SELECT uuid, class_name, record_name, entity_type, source,
                    source_path, game_version, properties_json
             FROM entities WHERE uuid = ?1"
        )?;

        let node = stmt.query_row(params![uuid.to_string()], |row| {
            Ok(row_to_node(row))
        }).ok();

        Ok(node)
    }

    pub fn lookup_by_class_name(&self, class_name: &str) -> Result<Vec<Node>> {
        let mut stmt = self.conn.prepare(
            "SELECT uuid, class_name, record_name, entity_type, source,
                    source_path, game_version, properties_json
             FROM entities WHERE class_name = ?1"
        )?;

        let nodes = stmt.query_map(params![class_name], |row| {
            Ok(row_to_node(row))
        })?.filter_map(|r| r.ok()).collect();

        Ok(nodes)
    }

    pub fn filter_by_type(&self, entity_type: EntityType, limit: usize) -> Result<Vec<Node>> {
        let mut stmt = self.conn.prepare(
            "SELECT uuid, class_name, record_name, entity_type, source,
                    source_path, game_version, properties_json
             FROM entities WHERE entity_type = ?1 LIMIT ?2"
        )?;

        let nodes = stmt.query_map(params![entity_type.as_str(), limit as i64], |row| {
            Ok(row_to_node(row))
        })?.filter_map(|r| r.ok()).collect();

        Ok(nodes)
    }

    pub fn execute_raw(&self, sql: &str) -> Result<Vec<Vec<String>>> {
        let mut stmt = self.conn.prepare(sql)?;
        let col_count = stmt.column_count();
        let rows = stmt.query_map([], |row| {
            let mut vals = Vec::new();
            for i in 0..col_count {
                let val: String = row.get::<_, String>(i).unwrap_or_default();
                vals.push(val);
            }
            Ok(vals)
        })?.filter_map(|r| r.ok()).collect();

        Ok(rows)
    }

    pub fn add_version(&self, code: &str, build_number: Option<&str>, data_path: &str) -> Result<()> {
        self.conn.execute(
            "INSERT OR REPLACE INTO versions (code, build_number, data_path) VALUES (?1, ?2, ?3)",
            params![code, build_number, data_path],
        )?;
        Ok(())
    }

    pub fn list_versions(&self) -> Result<Vec<VersionInfo>> {
        let mut stmt = self.conn.prepare(
            "SELECT id, code, build_number, data_path FROM versions"
        )?;
        let versions = stmt.query_map([], |row| {
            Ok(VersionInfo {
                id: row.get(0)?,
                code: row.get(1)?,
                build_number: row.get(2)?,
                data_path: row.get(3)?,
            })
        })?.filter_map(|r| r.ok()).collect();

        Ok(versions)
    }

    pub fn drop_version(&self, version: &str) -> Result<()> {
        self.conn.execute("DELETE FROM entities WHERE game_version = ?1", params![version])?;
        self.conn.execute("DELETE FROM versions WHERE code = ?1", params![version])?;
        Ok(())
    }

    pub fn category_counts(&self, version: &str) -> Result<Vec<(String, i64)>> {
        let mut stmt = self.conn.prepare(
            "SELECT entity_type, COUNT(*) FROM entities WHERE game_version = ?1 GROUP BY entity_type ORDER BY COUNT(*) DESC"
        )?;
        let counts = stmt.query_map(params![version], |row| {
            Ok((row.get::<_, String>(0)?, row.get::<_, i64>(1)?))
        })?.filter_map(|r| r.ok()).collect();

        Ok(counts)
    }
}

fn row_to_node(row: &rusqlite::Row) -> Node {
    let uuid_str: String = row.get(0).unwrap_or_default();
    let props_json: String = row.get(7).unwrap_or_else(|_| "{}".to_string());
    let entity_type_str: String = row.get(3).unwrap_or_else(|_| "Unknown".to_string());

    Node {
        id: uuid::Uuid::parse_str(&uuid_str).unwrap_or(uuid::Uuid::nil()),
        class_name: row.get(1).unwrap_or_default(),
        record_name: row.get(2).unwrap_or_default(),
        entity_type: entity_type_str.parse().unwrap_or(EntityType::Unknown),
        source: row.get(4).unwrap_or_default(),
        source_path: row.get(5).unwrap_or_default(),
        game_version: row.get(6).unwrap_or_default(),
        properties: serde_json::from_str(&props_json).unwrap_or_default(),
    }
}
```

```rust
use anyhow::{Context, Result};
use cozo::*;
use std::collections::BTreeMap;
use std::path::Path;

pub struct CozoGraph {
    db: DbInstance,
}

impl CozoGraph {
    pub fn open(path: &Path) -> Result<Self> {
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)?;
        }
        let db = DbInstance::new("sqlite", path.to_string_lossy().as_ref(), "")
            .map_err(|e| anyhow::anyhow!("Failed to open Cozo: {e}"))?;

        let graph = Self { db };
        graph.create_schema()?;
        Ok(graph)
    }

    pub fn open_in_memory() -> Result<Self> {
        let db = DbInstance::new("mem", "", "")
            .map_err(|e| anyhow::anyhow!("Failed to open Cozo in memory: {e}"))?;
        let graph = Self { db };
        graph.create_schema()?;
        Ok(graph)
    }

    fn create_schema(&self) -> Result<()> {
        // Create stored relations
        self.run_query(":create entity {uuid: String => class_name: String, entity_type: String, version: String}")?;
        self.run_query(":create edge {src: String, dst: String => label: String, source_field: String}")?;
        Ok(())
    }

    fn run_query(&self, query: &str) -> Result<NamedRows> {
        self.db.run_script(query, Default::default(), ScriptMutability::Mutable)
            .map_err(|e| anyhow::anyhow!("Cozo query error: {e}"))
    }

    fn run_immutable(&self, query: &str) -> Result<NamedRows> {
        self.db.run_script(query, Default::default(), ScriptMutability::Immutable)
            .map_err(|e| anyhow::anyhow!("Cozo query error: {e}"))
    }

    pub fn entity_count(&self) -> Result<usize> {
        let result = self.run_immutable("?[count(uuid)] := *entity{uuid}")?;
        if let Some(row) = result.rows.first() {
            if let Some(DataValue::Num(n)) = row.first() {
                return Ok(n.get_int().unwrap_or(0) as usize);
            }
        }
        Ok(0)
    }

    pub fn insert_entities(&self, entities: &[(uuid::Uuid, &str, &str, &str)]) -> Result<()> {
        if entities.is_empty() { return Ok(()); }

        let mut rows = String::from("?[uuid, class_name, entity_type, version] <- [");
        for (i, (uuid, class_name, entity_type, version)) in entities.iter().enumerate() {
            if i > 0 { rows.push_str(", "); }
            rows.push_str(&format!(
                "[\"{uuid}\", \"{class_name}\", \"{entity_type}\", \"{version}\"]"
            ));
        }
        rows.push_str("] :put entity {uuid => class_name, entity_type, version}");
        self.run_query(&rows)?;
        Ok(())
    }

    pub fn insert_edges(&self, edges: &[(uuid::Uuid, uuid::Uuid, &str, &str)]) -> Result<()> {
        if edges.is_empty() { return Ok(()); }

        let mut rows = String::from("?[src, dst, label, source_field] <- [");
        for (i, (src, dst, label, field)) in edges.iter().enumerate() {
            if i > 0 { rows.push_str(", "); }
            rows.push_str(&format!(
                "[\"{src}\", \"{dst}\", \"{label}\", \"{field}\"]"
            ));
        }
        rows.push_str("] :put edge {src, dst => label, source_field}");
        self.run_query(&rows)?;
        Ok(())
    }

    /// Traverse from a node up to `depth` hops, returning all reachable entity UUIDs.
    pub fn traverse(&self, start: &uuid::Uuid, depth: u32) -> Result<Vec<String>> {
        let query = format!(
            r#"
            start[uuid] <- [["{start}"]]
            reachable[dst] := start[src], *edge{{src, dst}}
            reachable[dst] := reachable[src], *edge{{src, dst}}
            ?[uuid] := reachable[uuid]
            :limit {depth_limit}
            "#,
            start = start,
            depth_limit = depth * 100, // limit total results, not depth
        );

        // Use iterative approach for depth control
        let mut query_parts = vec![format!(r#"start[uuid] <- [["{start}"]]"#)];

        // Build N-hop reachable relation
        query_parts.push("hop_1[dst] := start[src], *edge{src, dst}".to_string());
        for i in 2..=depth {
            query_parts.push(format!(
                "hop_{i}[dst] := hop_{}[src], *edge{{src, dst}}", i - 1
            ));
        }

        // Union all hops
        let mut union_parts = Vec::new();
        for i in 1..=depth {
            union_parts.push(format!("reachable[uuid] := hop_{i}[uuid]"));
        }
        query_parts.extend(union_parts);
        query_parts.push("?[uuid] := reachable[uuid]".to_string());

        let full_query = query_parts.join("\n");
        let result = self.run_immutable(&full_query)?;

        let uuids = result.rows.iter()
            .filter_map(|row| {
                if let Some(DataValue::Str(s)) = row.first() {
                    Some(s.to_string())
                } else {
                    None
                }
            })
            .collect();

        Ok(uuids)
    }

    /// Find the shortest path between two entities (BFS up to max_depth).
    pub fn find_path(&self, from: &uuid::Uuid, to: &uuid::Uuid, max_depth: u32) -> Result<Option<Vec<String>>> {
        // Cozo doesn't have built-in BFS path, so we do iterative widening
        for depth in 1..=max_depth {
            let mut query_parts = vec![
                format!(r#"start[uuid, path] <- [["{from}", ["{from}"]]]"#),
            ];

            for i in 1..=depth {
                let prev = if i == 1 { "start" } else { &format!("hop_{}", i - 1) };
                query_parts.push(format!(
                    r#"hop_{i}[dst, path] := {prev}[src, prev_path], *edge{{src, dst}}, path = append(prev_path, [dst])"#
                ));
            }

            // Check if target is reachable at any depth
            let mut found_parts = Vec::new();
            for i in 1..=depth {
                found_parts.push(format!(
                    r#"found[path] := hop_{i}[uuid, path], uuid = "{to}""#
                ));
            }
            query_parts.extend(found_parts);
            query_parts.push("?[path] := found[path] :limit 1".to_string());

            let full_query = query_parts.join("\n");
            match self.run_immutable(&full_query) {
                Ok(result) => {
                    if let Some(row) = result.rows.first() {
                        if let Some(DataValue::List(path)) = row.first() {
                            let path_strs: Vec<String> = path.iter()
                                .filter_map(|v| {
                                    if let DataValue::Str(s) = v {
                                        Some(s.to_string())
                                    } else {
                                        None
                                    }
                                })
                                .collect();
                            if !path_strs.is_empty() {
                                return Ok(Some(path_strs));
                            }
                        }
                    }
                }
                Err(_) => continue, // Datalog syntax may fail on certain depths, try next
            }
        }

        Ok(None)
    }

    /// Execute a raw Datalog query.
    pub fn execute_raw(&self, query: &str) -> Result<NamedRows> {
```

```rust
pub mod sqlite;
pub mod cozo;

use crate::config::Config;
use crate::model::*;
use crate::parser;
use anyhow::{Context, Result};
use std::path::Path;

pub struct Indexer {
    pub sqlite: sqlite::SqliteIndex,
    pub graph: cozo::CozoGraph,
}

impl Indexer {
    /// Open or create the index at the configured path.
    pub fn open(config: &Config) -> Result<Self> {
        let index_dir = Path::new(&config.index.path);
        std::fs::create_dir_all(index_dir)?;

        let sqlite = sqlite::SqliteIndex::open(&index_dir.join("entities.db"))
            .context("opening SQLite index")?;
        let graph = cozo::CozoGraph::open(&index_dir.join("graph.db"))
            .context("opening Cozo graph")?;

        Ok(Self { sqlite, graph })
    }

    /// Index a data directory. Detects parsers automatically.
    pub fn index_directory(&self, data_path: &str, version: &str, parser_name: &str) -> Result<IndexStats> {
        let path = Path::new(data_path);
        anyhow::ensure!(path.is_dir(), "Data path does not exist: {data_path}");

        // Select parsers
        let parsers: Vec<Box<dyn parser::P4kParser>> = if parser_name == "auto" {
            let detected = parser::detect_parsers(path);
            if detected.is_empty() {
                anyhow::bail!("No parser detected for: {data_path}");
            }
            detected
        } else {
            parser::all_parsers()
                .into_iter()
                .filter(|p| p.name() == parser_name)
                .collect()
        };

        tracing::info!("Indexing {data_path} as {version} with {} parser(s)", parsers.len());

        let mut total_result = ParseResult::new();

        for p in &parsers {
            tracing::info!("Running {} parser...", p.name());
            let result = p.parse(path, version)
                .with_context(|| format!("{} parser failed", p.name()))?;
            tracing::info!(
                "{} parser: {} nodes, {} edges, {} warnings",
                p.name(), result.nodes.len(), result.edges.len(), result.warnings.len()
            );
            total_result.merge(result);
        }

        // Store in SQLite
        tracing::info!("Inserting {} nodes into SQLite...", total_result.nodes.len());
        self.sqlite.insert_nodes(&total_result.nodes)?;
        self.sqlite.add_version(version, None, data_path)?;

        // Store in Cozo graph
        tracing::info!("Inserting {} entities into graph...", total_result.nodes.len());
        let entities: Vec<(uuid::Uuid, &str, &str, &str)> = total_result.nodes.iter()
            .map(|n| (n.id, n.class_name.as_str(), n.entity_type.as_str(), n.game_version.as_str()))
            .collect();

        // Batch insert in chunks of 500
        for chunk in entities.chunks(500) {
            self.graph.insert_entities(chunk)?;
        }

        tracing::info!("Inserting {} edges into graph...", total_result.edges.len());
        let edges: Vec<(uuid::Uuid, uuid::Uuid, &str, &str)> = total_result.edges.iter()
            .map(|e| (e.source_id, e.target_id, e.label.as_str(), e.source_field.as_str()))
            .collect();

        for chunk in edges.chunks(500) {
            self.graph.insert_edges(chunk)?;
        }

        let stats = IndexStats {
            version: version.to_string(),
            node_count: total_result.nodes.len(),
            edge_count: total_result.edges.len(),
            warning_count: total_result.warnings.len(),
            parsers_used: parsers.iter().map(|p| p.name().to_string()).collect(),
        };

        tracing::info!("Indexing complete: {stats:?}");
        Ok(stats)
    }

    /// Drop all data for a version and re-index.
    pub fn reindex(&self, data_path: &str, version: &str, parser_name: &str) -> Result<IndexStats> {
        tracing::info!("Dropping existing index for {version}...");
        self.sqlite.drop_version(version)?;
        self.graph.drop_version(version)?;
        self.index_directory(data_path, version, parser_name)
    }
}

#[derive(Debug)]
pub struct IndexStats {
    pub version: String,
    pub node_count: usize,
    pub edge_count: usize,
    pub warning_count: usize,
    pub parsers_used: Vec<String>,
}
```

```rust
use crate::index::Indexer;
use crate::model::*;
use anyhow::Result;

pub struct QueryEngine<'a> {
    indexer: &'a Indexer,
}

#[derive(Debug)]
pub struct StatusInfo {
    pub entity_count: i64,
    pub versions: Vec<crate::index::sqlite::VersionInfo>,
    pub category_counts: Vec<(String, i64)>,
}

impl<'a> QueryEngine<'a> {
    pub fn new(indexer: &'a Indexer) -> Self {
        Self { indexer }
    }

    /// Keyword search (routes to SQLite FTS5).
    pub fn search(&self, query: &str, limit: usize) -> Result<Vec<Node>> {
        self.indexer.sqlite.search(query, limit)
    }

    /// Lookup by UUID (routes to SQLite).
    pub fn lookup_by_uuid(&self, uuid: &uuid::Uuid) -> Result<Option<Node>> {
        self.indexer.sqlite.lookup_by_uuid(uuid)
    }

    /// Lookup by class_name (routes to SQLite).
    pub fn lookup_by_class_name(&self, class_name: &str) -> Result<Vec<Node>> {
        self.indexer.sqlite.lookup_by_class_name(class_name)
    }

    /// Filter by entity type (routes to SQLite).
    pub fn filter_by_type(&self, entity_type: EntityType, limit: usize) -> Result<Vec<Node>> {
        self.indexer.sqlite.filter_by_type(entity_type, limit)
    }

    /// Traverse relationships from an entity (routes to Cozo).
    pub fn traverse(&self, uuid: &uuid::Uuid, depth: u32) -> Result<Vec<Node>> {
        let related_uuids = self.indexer.graph.traverse(uuid, depth)?;
        let mut nodes = Vec::new();
        for uuid_str in related_uuids {
            if let Ok(uuid) = uuid::Uuid::parse_str(&uuid_str) {
                if let Some(node) = self.indexer.sqlite.lookup_by_uuid(&uuid)? {
                    nodes.push(node);
                }
            }
        }
        Ok(nodes)
    }

    /// Find path between two entities (routes to Cozo).
    pub fn find_path(&self, from: &uuid::Uuid, to: &uuid::Uuid, max_depth: u32) -> Result<Option<Vec<Node>>> {
        let path_uuids = self.indexer.graph.find_path(from, to, max_depth)?;
        match path_uuids {
            Some(uuids) => {
                let mut nodes = Vec::new();
                for uuid_str in uuids {
                    if let Ok(uuid) = uuid::Uuid::parse_str(&uuid_str) {
                        if let Some(node) = self.indexer.sqlite.lookup_by_uuid(&uuid)? {
                            nodes.push(node);
                        }
                    }
                }
                Ok(Some(nodes))
            }
            None => Ok(None),
        }
    }

    /// Execute raw SQL against the SQLite index.
    pub fn raw_sql(&self, sql: &str) -> Result<Vec<Vec<String>>> {
        self.indexer.sqlite.execute_raw(sql)
    }

    /// Execute raw Datalog against the Cozo graph.
    pub fn raw_datalog(&self, query: &str) -> Result<String> {
        let result = self.indexer.graph.execute_raw(query)?;
        // Format as JSON for the LLM
        let rows: Vec<Vec<String>> = result.rows.iter()
            .map(|row| {
                row.iter().map(|v| format!("{v:?}")).collect()
            })
            .collect();
        Ok(serde_json::to_string_pretty(&serde_json::json!({
            "headers": result.headers,
            "rows": rows,
        }))?)
    }

    /// Get index status.
    pub fn status(&self) -> Result<StatusInfo> {
        let entity_count = self.indexer.sqlite.entity_count()?;
        let versions = self.indexer.sqlite.list_versions()?;
        let category_counts = if let Some(v) = versions.first() {
            self.indexer.sqlite.category_counts(&v.code)?
        } else {
            vec![]
        };

        Ok(StatusInfo {
            entity_count,
            versions,
            category_counts,
        })
    }
}
```

```rust
        if let Ok(mut config) = self.config.write() {
            config.add_version(&req.path, &version, "auto");
            let _ = config.save();
        }

        match result {
            Ok(stats) => Ok(CallToolResult::success(vec![Content::text(
                format!("Indexed {}: {} entities, {} edges, {} warnings (parsers: {})",
                    stats.version, stats.node_count, stats.edge_count,
                    stats.warning_count, stats.parsers_used.join(", "))
            )])),
            Err(e) => Ok(CallToolResult::error(vec![Content::text(format!("Index error: {e}"))]))
        }
    }

    #[tool(
        name = "status",
        description = "Show indexed game versions, entity counts, and index health"
    )]
    fn status(&self) -> Result<CallToolResult, McpError> {
        let engine = self.query_engine();
        match engine.status() {
            Ok(info) => {
                let mut text = format!("Entity count: {}\n\nVersions:\n", info.entity_count);
                for v in &info.versions {
                    text.push_str(&format!("  - {} (build: {}, path: {})\n",
                        v.code, v.build_number.as_deref().unwrap_or("n/a"), v.data_path));
                }
                if !info.category_counts.is_empty() {
                    text.push_str("\nCategories:\n");
                    for (cat, count) in &info.category_counts {
                        text.push_str(&format!("  {cat}: {count}\n"));
                    }
                }
                Ok(CallToolResult::success(vec![Content::text(text)]))
            }
            Err(e) => Ok(CallToolResult::error(vec![Content::text(format!("Status error: {e}"))]))
        }
    }

    #[tool(
        name = "diff",
        description = "Compare an entity between two game versions — shows what changed"
    )]
    fn diff(&self, Parameters(req): Parameters<DiffRequest>) -> Result<CallToolResult, McpError> {
        let engine = self.query_engine();

        // Find entity in both versions
        let nodes = engine.lookup_by_class_name(&req.entity)
            .map_err(|e| McpError::new(ErrorCode::INTERNAL_ERROR, format!("{e}"), None))?;

        let node_a = nodes.iter().find(|n| n.game_version == req.version_a);
        let node_b = nodes.iter().find(|n| n.game_version == req.version_b);

        match (node_a, node_b) {
            (Some(a), Some(b)) => {
                let props_a = serde_json::to_string_pretty(&a.properties).unwrap_or_default();
                let props_b = serde_json::to_string_pretty(&b.properties).unwrap_or_default();
                Ok(CallToolResult::success(vec![Content::text(
                    format!("## {} — {ver_a} vs {ver_b}\n\n### {ver_a}\n```json\n{props_a}\n```\n\n### {ver_b}\n```json\n{props_b}\n```",
                        a.class_name, ver_a = req.version_a, ver_b = req.version_b)
                )]))
            }
            (Some(_), None) => Ok(CallToolResult::success(vec![Content::text(
                format!("Entity exists in {} but not in {}", req.version_a, req.version_b)
            )])),
            (None, Some(_)) => Ok(CallToolResult::success(vec![Content::text(
                format!("Entity exists in {} but not in {}", req.version_b, req.version_a)
            )])),
            (None, None) => Ok(CallToolResult::success(vec![Content::text(
                format!("Entity '{}' not found in either version", req.entity)
            )])),
        }
    }

    #[tool(
        name = "locate",
        description = "Find where an entity exists in the game world (SOC locations)"
    )]
    fn locate(&self, Parameters(req): Parameters<LocateRequest>) -> Result<CallToolResult, McpError> {
        // Try UUID first, then class_name
        let engine = self.query_engine();
        let uuid = uuid::Uuid::parse_str(&req.entity).ok();

        let target_uuid = if let Some(u) = uuid {
            u
        } else {
            // Look up by class_name
            match engine.lookup_by_class_name(&req.entity) {
                Ok(nodes) if !nodes.is_empty() => nodes[0].id,
                _ => return Ok(CallToolResult::success(vec![Content::text("Entity not found")]))
            }
        };

        // Traverse looking for Location-type entities
        match engine.traverse(&target_uuid, 3) {
            Ok(nodes) => {
                let locations: Vec<_> = nodes.iter()
                    .filter(|n| n.entity_type == crate::model::EntityType::Location)
                    .collect();
                if locations.is_empty() {
                    Ok(CallToolResult::success(vec![Content::text(
                        "No game world locations found for this entity within 3 hops"
                    )]))
                } else {
                    Ok(CallToolResult::success(vec![Content::text(format_nodes(&locations.into_iter().cloned().collect::<Vec<_>>()))]))
                }
            }
            Err(e) => Ok(CallToolResult::error(vec![Content::text(format!("Locate error: {e}"))]))
        }
    }

    #[tool(
        name = "who_uses",
        description = "Find which NPCs, loadouts, or ships reference a specific item"
    )]
    fn who_uses(&self, Parameters(req): Parameters<WhoUsesRequest>) -> Result<CallToolResult, McpError> {
        let engine = self.query_engine();
        let uuid = uuid::Uuid::parse_str(&req.item).ok();

        let target_uuid = if let Some(u) = uuid {
            u
        } else {
            match engine.lookup_by_class_name(&req.item) {
                Ok(nodes) if !nodes.is_empty() => nodes[0].id,
                _ => return Ok(CallToolResult::success(vec![Content::text("Item not found")]))
            }
        };

        // Traverse looking for NPC/Loadout/Ship-type entities
        match engine.traverse(&target_uuid, 3) {
            Ok(nodes) => {
                let users: Vec<_> = nodes.iter()
                    .filter(|n| matches!(n.entity_type,
                        crate::model::EntityType::NPC
                        | crate::model::EntityType::Loadout
                        | crate::model::EntityType::Ship))
                    .collect();
                if users.is_empty() {
                    Ok(CallToolResult::success(vec![Content::text(
                        "No NPCs, loadouts, or ships found using this item within 3 hops"
                    )]))
                } else {
                    Ok(CallToolResult::success(vec![Content::text(format_nodes(&users.into_iter().cloned().collect::<Vec<_>>()))]))
                }
            }
            Err(e) => Ok(CallToolResult::error(vec![Content::text(format!("who_uses error: {e}"))]))
        }
    }
}

fn format_nodes(nodes: &[crate::model::Node]) -> String {
    if nodes.is_empty() {
        return "No results".to_string();
    }
    nodes.iter()
        .map(|n| format!("[{type}] {name} — uuid: {uuid} (source: {source})",
            type = n.entity_type,
            name = n.class_name,
            uuid = n.id,
            source = n.source))
        .collect::<Vec<_>>()
        .join("\n")
}

fn format_nodes_detailed(nodes: &[crate::model::Node]) -> String {
    nodes.iter()
        .map(|n| {
            let props = serde_json::to_string_pretty(&n.properties).unwrap_or_default();
            format!(
                "## {name}\n- Type: {type}\n- UUID: {uuid}\n- Record: {record}\n- Source: {source} ({path})\n- Version: {version}\n- Properties:\n```json\n{props}\n```",
                name = n.class_name,
                type = n.entity_type,
                uuid = n.id,
                record = n.record_name,
                source = n.source,
                path = n.source_path,
                version = n.game_version,
            )
        })
        .collect::<Vec<_>>()
        .join("\n\n---\n\n")
}
```
